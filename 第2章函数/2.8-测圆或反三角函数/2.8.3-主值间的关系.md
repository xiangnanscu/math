# 2.8.3 主值间的关系
$$
\arcsin x = \frac{\pi }{2} - \arccos x = \arctan \frac{x}{\sqrt{1 - {x}^{2}}}
$$
$$
= \left\{  \begin{array}{ll}  - \arccos \sqrt{1 - {x}^{2}} & \left( {-1 \leq  x \leq  0}\right) , \\  \arccos \sqrt{1 - {x}^{2}} & \left( {0 \leq  x \leq  1}\right) , \end{array}\right.  \tag{2.145}
$$
$$
\arccos x = \frac{\pi }{2} - \arcsin x = \operatorname{arccot}\frac{x}{\sqrt{1 - {x}^{2}}}
$$
$$
= \left\{  \begin{array}{ll} \pi  - \arcsin \sqrt{1 - {x}^{2}} & \left( {\pi  - 1 \leq  x \leq  0}\right) , \\  \arcsin \sqrt{1 - {x}^{2}} & \left( {0 \leq  x \leq  1}\right) , \end{array}\right.  \tag{2.146}
$$
$$
\arctan x = \frac{\pi }{2} - \operatorname{arccot}x = \arcsin \frac{x}{\sqrt{1 + {x}^{2}}}, \tag{2.147}
$$
$$
\arctan x = \left\{  {\begin{array}{ll} \operatorname{arccot}\frac{1}{x} - \pi & \left( {x < 0}\right) \\  \operatorname{arccot}\frac{1}{x} & \left( {x > 0}\right)  \end{array} = \left\{  \begin{array}{ll}  - \arccos \frac{1}{\sqrt{1 + {x}^{2}}} & \left( {x \leq  0}\right) , \\  \arccos \frac{1}{\sqrt{1 + {x}^{2}}} & \left( {x \geq  0}\right) , \end{array}\right. }\right.
$$
(2.148)
$$
\operatorname{arccot}x = \frac{\pi }{2} - \arctan x = \arccos \frac{x}{\sqrt{1 + {x}^{2}}}, \tag{2.149}
$$
$$
\operatorname{arccot}x = \left\{  {\begin{array}{ll} \arctan \frac{1}{x} + \pi & \left( {x < 0}\right) \\  \arctan \frac{1}{x} & \left( {x > 0}\right)  \end{array} = \left\{  \begin{array}{ll} \pi  - \arcsin \frac{1}{\sqrt{1 + {x}^{2}}} & \left( {x \leq  0}\right) , \\  \arcsin \frac{1}{\sqrt{1 + {x}^{2}}} & \left( {x \geq  0}\right) . \end{array}\right. }\right.
$$
